노트북 기준 f_video_chat_rag.ipynb

DB에서 tbl_ai_video_summary와 tbl_video_session을 조인해 문서를 만들고, FAISS 검색과 Redis answer cache를 사용한다.

캐시 scope는 member_id와 선택적 video_session_id로 만들어야 한다. Spring에서 인증된 memberId를 넣고, 브라우저 입력값으로 받지 않는다.

실제 DB에는 tbl_ai_video_summary.video_session_id의 FK와 UNIQUE가 있으므로, repo DDL에도 같은 구조를 반영한다.

결론: 브라우저가 FastAPI를 직접 호출하면 member_id를 조작할 수 있다. Spring이 인증된 사용자에서 memberId를 꺼내 FastAPI로 내부 호출하고, FastAPI는 cache lookup보다 먼저 member_id + video_session_id 접근 권한을 검증한다.

class Database:
    def __init__(self):
        self._pool = None

    async def connect(self):
        if self._pool is None:
            self._pool = await asyncpg.create_pool(settings.database_url)

    async def disconnect(self):
        if self._pool is not None:
            await self._pool.close()

    @asynccontextmanager
    async def get_connection(self):
        async with self._pool.acquire() as conn:
            yield conn

from app.db.database import db

class VideoSummaryRepository:
    async def member_can_access_video_session(self, member_id: int, video_session_id: int) -> bool:
        query = """
            select exists (
                select 1
                from tbl_video_session
                where id = $2
                  and (caller_id = $1 or receiver_id = $1)
            )
        """
        async with db.get_connection() as conn:
            return await conn.fetchval(query, member_id, video_session_id)

    async def find_summaries_for_member(self, member_id: int, video_session_id: int | None = None):
        query = """
            select
                s.id as summary_id,
                s.video_session_id,
                vs.conversation_id,
                vs.caller_id,
                vs.receiver_id,
                s.created_datetime,
                s.summary
            from tbl_ai_video_summary s
            join tbl_video_session vs on vs.id = s.video_session_id
            where (vs.caller_id = $1 or vs.receiver_id = $1)
              and ($2::bigint is null or s.video_session_id = $2)
            order by s.video_session_id
        """
        async with db.get_connection() as conn:
            rows = await conn.fetch(query, member_id, video_session_id)
            return [dict(row) for row in rows]

    async def insert_summary(self, video_session_id: int, summary: str) -> int:
        query = """
            insert into tbl_ai_video_summary (video_session_id, summary)
            values ($1, $2)
            on conflict (video_session_id)
            do update set summary = excluded.summary
            returning id
        """
        async with db.get_connection() as conn:
            return await conn.fetchval(query, video_session_id, summary)

    async def insert_query_log(self, member_id: int, question: str, answer: str, cache_hit: bool):
        query = """
            insert into tbl_ai_video_chat_query_log
                (member_id, question, answer, cache_hit)
            values ($1, $2, $3, $4)
        """
        async with db.get_connection() as conn:
            await conn.execute(query, member_id, question, answer, cache_hit)
app/service/semantic_cache.py노트북의 Redis cache scope 유지
import json
from langchain.globals import set_llm_cache
from langchain_huggingface import HuggingFaceEmbeddings
from langchain_community.vectorstores import Redis
from langchain_community.vectorstores.redis import RedisTag
from app.core.config import settings

set_llm_cache(None)

cache_embeddings = HuggingFaceEmbeddings(
    model_name="jhgan/ko-sbert-nli",
    model_kwargs={"device": "cpu"},
    encode_kwargs={"normalize_embeddings": True},
)

ANSWER_CACHE_SCHEMA = {
    "text": [{"name": "question"}],
    "tag": [{"name": "answer"}, {"name": "scope"}],
}

def build_cache_scope(member_id: int, video_session_id: int | None = None) -> str:
    if member_id is None:
        raise ValueError("member_id is required")
    session_scope = "all" if video_session_id is None else str(video_session_id)
    return f"member_{member_id}__video_session_{session_scope}"

def build_cache_query(question: str, member_id: int, video_session_id: int | None = None) -> str:
    return f"{build_cache_scope(member_id, video_session_id)}\n질문: {question}"

class SemanticAnswerCache:
    def _store(self):
        return Redis.from_existing_index(
            cache_embeddings,
            index_name=settings.answer_cache_index,
            redis_url=settings.redis_url,
            schema=ANSWER_CACHE_SCHEMA,
        )

    def lookup(self, question: str, member_id: int, video_session_id: int | None = None):
        scope = build_cache_scope(member_id, video_session_id)
        try:
            results = self._store().similarity_search_with_score(
                build_cache_query(question, member_id, video_session_id),
                k=1,
                filter=RedisTag("scope") == scope,
            )
        except Exception:
            return None

        if not results:
            return None
        doc, score = results[0]
        if score <= settings.answer_cache_score_threshold and doc.metadata.get("scope") == scope:
            return {"answer": doc.metadata["answer"], "score": score, "scope": scope}
        return None

    def store(self, question: str, answer: str, member_id: int, video_session_id: int | None = None):
        scope = build_cache_scope(member_id, video_session_id)
        Redis.from_texts(
            texts=[build_cache_query(question, member_id, video_session_id)],
            embedding=cache_embeddings,
            metadatas=[{"question": question, "answer": answer, "scope": scope}],
            index_name=settings.answer_cache_index,
            redis_url=settings.redis_url,
            index_schema=ANSWER_CACHE_SCHEMA,
        )

def sse_json(payload: dict) -> str:
    return "data: " + json.dumps(payload, ensure_ascii=False) + "\n\n"
app/service/video_rag_service.pyRAG miss일 때만 LLM 호출
import asyncio
from langchain_core.documents import Document
from langchain_community.vectorstores import FAISS
from langchain_huggingface import HuggingFaceEmbeddings
from langchain_openai import ChatOpenAI
from langchain_core.prompts import ChatPromptTemplate
from langchain_core.output_parsers import StrOutputParser
from app.repository.video_summary_repository import VideoSummaryRepository
from app.service.semantic_cache import SemanticAnswerCache, sse_json

class VideoRagService:
    def __init__(self):
        self.repo = VideoSummaryRepository()
        self.cache = SemanticAnswerCache()
        self.embeddings = HuggingFaceEmbeddings(
            model_name="jhgan/ko-sbert-nli",
            model_kwargs={"device": "cpu"},
            encode_kwargs={"normalize_embeddings": True},
        )
        self.llm = ChatOpenAI(model_name="gpt-5.4-mini", temperature=0, cache=False)
        self.prompt = ChatPromptTemplate.from_template(
            "너는 사용자의 화상채팅 요약 기록만 근거로 답한다.\n"
            "근거가 없으면 모른다고 답한다.\n\n"
            "[화상채팅 요약]\n{context}\n\n[질문]\n{question}"
        )
        self.chain = self.prompt | self.llm | StrOutputParser()

    async def assert_authorized(self, member_id: int, video_session_id: int | None):
        if member_id is None:
            raise PermissionError("member_id is required")
        if video_session_id is None:
            return
        allowed = await self.repo.member_can_access_video_session(member_id, video_session_id)
        if not allowed:
            raise PermissionError("video_session_id is not accessible for member_id")

    async def _build_member_vectorstore(self, member_id: int, video_session_id: int | None):
        rows = await self.repo.find_summaries_for_member(member_id, video_session_id)
        docs = [
            Document(
                page_content=row["summary"],
                metadata={
                    "summary_id": row["summary_id"],
                    "video_session_id": row["video_session_id"],
                    "conversation_id": row["conversation_id"],
                    "caller_id": row["caller_id"],
                    "receiver_id": row["receiver_id"],
                },
            )
            for row in rows
        ]
        if not docs:
            return None
        return FAISS.from_documents(docs, self.embeddings)

    async def _retrieve_context(self, question: str, member_id: int, video_session_id: int | None):
        vectorstore = await self._build_member_vectorstore(member_id, video_session_id)
        if vectorstore is None:
            return ""
        docs = vectorstore.similarity_search(question, k=4)
        return "\n\n".join(
            f"[video_session_id={d.metadata['video_session_id']}]\n{d.page_content}"
            for d in docs
        )

    async def stream_answer(self, question: str, member_id: int, video_session_id: int | None = None):
        # router에서 assert_authorized를 먼저 호출한 뒤 StreamingResponse를 생성한다.
        hit = self.cache.lookup(question, member_id, video_session_id)
        if hit:
            yield sse_json({"type": "meta", "cacheHit": True, "score": hit["score"]})
            for char in hit["answer"]:
                yield sse_json({"type": "token", "text": char})
                await asyncio.sleep(0.005)
            yield sse_json({"type": "done"})
            return

        context = await self._retrieve_context(question, member_id, video_session_id)
        if not context:
            msg = "확인 가능한 화상채팅 요약 기록이 없습니다."
            yield sse_json({"type": "meta", "cacheHit": False})
            yield sse_json({"type": "token", "text": msg})
            yield sse_json({"type": "done"})
            return

        answer = ""
        yield sse_json({"type": "meta", "cacheHit": False})
        async for token in self.chain.astream({"context": context, "question": question}):
            answer += token
            yield sse_json({"type": "token", "text": token})

        self.cache.store(question, answer, member_id, video_session_id)
        await self.repo.insert_query_log(member_id, question, answer, cache_hit=False)
        yield sse_json({"type": "done"})

import lombok.RequiredArgsConstructor;
import org.springframework.beans.factory.annotation.Value;

import reactor.core.publisher.Flux;

@Component
@RequiredArgsConstructor
public class VideoChatAiClient {
    private final WebClient fastAiWebClient;

    @Value("${fast-ai.internal-token}")
    private String internalToken;

    public Flux<String> stream(VideoChatAiInternalRequest request) {
        return fastAiWebClient.post()
                .uri("/internal/ai/video-chat/rag/stream")
                .contentType(MediaType.APPLICATION_JSON)
                .accept(MediaType.TEXT_EVENT_STREAM)
                .header("X-AI-Internal-Token", internalToken)
                .bodyValue(request)
                .retrieve()
                .bodyToFlux(new ParameterizedTypeReference<ServerSentEvent<String>>() {})
                .map(ServerSentEvent::data);
    }
}
src/main/java/com/app/globalgates/service/video_chat_ai/VideoChatAiService.javaSseEmitter relay
package com.app.globalgates.service.video_chat_ai;

import com.app.globalgates.dto.video_chat_ai.VideoChatAiInternalRequest;
import com.app.globalgates.dto.video_chat_ai.VideoChatAiRequest;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;
import org.springframework.web.servlet.mvc.method.annotation.SseEmitter;

@Service
@RequiredArgsConstructor
public class VideoChatAiService {
    private final VideoChatAiClient client;

    public SseEmitter stream(Long memberId, VideoChatAiRequest request) {
        SseEmitter emitter = new SseEmitter(0L);
        VideoChatAiInternalRequest internalRequest = new VideoChatAiInternalRequest(
                request.question(),
                memberId,
                request.videoSessionId()
        );

        client.stream(internalRequest)
                .subscribe(
                        data -> send(emitter, data),
                        emitter::completeWithError,
                        emitter::complete
                );

        return emitter;
    }

    private void send(SseEmitter emitter, String data) {
        try {
            emitter.send(SseEmitter.event().data(data));
        } catch (Exception e) {
            emitter.completeWithError(e);
        }
    }
}

event.js의 더미 응답 교체

현재 src/main/resources/static/js/video_ai/event.js의 sendMessage()는 더미 메시지만 추가한다. 이 부분을 POST + response stream 처리로 바꾼다. EventSource는 GET 중심이므로, 질문 본문을 보내야 하는 현재 구조에서는 fetch()의 ReadableStream을 읽는 편이 맞다.

static/js/video_ai/event.jssendMessage 교체 방향
async function sendMessage() {
    const text = chatTextarea.value.trim();
    if (!text) return;

    addMessage(text, 'user');
    chatTextarea.value = '';
    chatTextarea.style.height = 'auto';

    const aiBubble = addMessage('', 'ai');

    const response = await fetch('/api/video-chat/ai/stream', {
        method: 'POST',
        credentials: 'same-origin',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
            question: text,
            videoSessionId: currentMeeting?.videoSessionId ?? null
        })
    });

    if (!response.ok || !response.body) {
        aiBubble.textContent = '답변을 가져오지 못했습니다.';
        return;
    }

    const reader = response.body.pipeThrough(new TextDecoderStream()).getReader();
    let buffer = '';

    while (true) {
        const { value, done } = await reader.read();
        if (done) break;
        buffer += value;

        const events = buffer.split('\n\n');
        buffer = events.pop() ?? '';

        for (const event of events) {
            const line = event.split('\n').find((part) => part.startsWith('data:'));
            if (!line) continue;

            const payload = JSON.parse(line.slice(5).trim());
            if (payload.type === 'token') {
                aiBubble.textContent += payload.text;
                chatMessages.scrollTop = chatMessages.scrollHeight;
            }
        }
    }
}
실제 적용 때는 addMessage()가 만든 DOM 노드를 반환하도록 바꿔야 한다. 지금 함수는 append만 하고 반환값이 없으므로, streaming 누적 렌더링을 하려면 그 부분도 같이 수정해야 한다.

rag.aquery(..., mode="hybrid") 사용 판단

현재 노트북과 LightRAG 가능성 분리

현재 노트북 기준

f_video_chat_rag.ipynb는 LangChain FAISS + Redis answer cache + LCEL chain 구조다. 이 상태를 FastAPI로 옮기면 chain.astream(...) 또는 llm.astream(...) 흐름이 자연스럽다.

따라서 위 FastAPI 코드는 현재 노트북 상태에 맞춰 self.chain.astream으로 작성했다.

LightRAG를 별도로 채택한다면

사용자가 제시한 rag.aquery(question, mode="hybrid")는 LightRAG 계열 API로 보인다. 최신 공식 소스의 QueryParam에는 mode와 stream 필드가 있다.

그 경우에는 아래처럼 FastAPI service 내부만 갈아끼우면 되고, Spring/Frontend 계약은 그대로 유지한다.

LightRAG 채택 시 FastAPI service 내부 대체 예시Spring 계약 유지
from lightrag.base import QueryParam
from app.service.semantic_cache import sse_json

async def stream_answer_with_lightrag(question: str, member_id: int, video_session_id: int | None = None):
    hit = cache.lookup(question, member_id, video_session_id)
    if hit:
        for char in hit["answer"]:
            yield sse_json({"type": "token", "text": char})
        yield sse_json({"type": "done"})
        return

    # 구현체에 따라 aquery 반환이 문자열 또는 async iterator일 수 있으므로 실제 설치 버전에서 확인 필요
    result = await rag.aquery(
        question,
        param=QueryParam(mode="hybrid", stream=True),
    )

    answer = ""
    async for chunk in result:
        answer += chunk
        yield sse_json({"type": "token", "text": chunk})

    cache.store(question, answer, member_id, video_session_id)
    yield sse_json({"type": "done"})
단, 현재 노트북에는 LightRAG 객체 초기화, storage 초기화, member별 데이터 격리 방식이 없다. 지금 단계에서 LightRAG로 갈아타면 “노트북 상태에 맞춘 이식”이 아니라 엔진 교체가 된다. 먼저 LangChain 버전으로 FastAPI를 붙이고, LightRAG는 별도 리팩터링 티켓으로 분리하는 편이 안전하다.

적용 순서

실패 지점을 작게 나누기

1단계

FastAPI에 member_id 필수화와 video_session_id 객체 권한 검증을 먼저 추가한다. cache lookup과 RAG 호출은 이 검증 뒤에만 실행한다.

2단계

FastAPI에 /internal/ai/video-chat/rag/stream 추가. member 범위 DB SELECT와 Redis cache를 붙이고 Spring 없이 curl/http 파일로 검증한다.

3단계

Spring에 WebClient client/service/controller 추가. 브라우저는 아직 건드리지 말고 Spring endpoint가 FastAPI stream을 relay하는지만 확인한다.

4단계

video_ai/event.js의 더미 응답을 fetch stream으로 교체한다. 이때 memberId는 절대 JS에서 보내지 않는다.

5단계

녹음 업로드 이후 STT/요약 INSERT와 reindex 트리거를 붙인다. 챗봇 Q&A가 안정화된 뒤 넣는다.

우선순위: member scope 보안 → 질문 스트림 연결 → 캐시 hit/miss 관측 → 요약 생성 INSERT 자동화.
